1. JavaScript คืออะไร?

JavaScript (JS) เป็นภาษาโปรแกรมแบบ interpreter, dynamic typing, single‑thread พร้อม event loop ใช้สำหรับสร้างเว็บแบบไดนามิก ควบคุม DOM, จัดการ event, โต้ตอบ backend (Node.js) มาตรฐาน ECMAScript (ES6 → ESNext)

คุณสมบัติรายละเอียด
สร้างโดยBrendan Eich (1995, Netscape)
มาตรฐานECMAScript (ES1 – ES2025)
ลักษณะเด่นJIT compiled, first‑class functions, prototype‑based OOP, async non‑blocking

19. 🔥 JavaScript Pro Tricks & Patterns
⚡ Debounce (จำกัดการเรียกฟังก์ชัน)
function debounce(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}
window.addEventListener('resize', debounce(() => console.log('resized'), 300));
⚡ Throttle (จำกัดความถี่)
function throttle(fn, limit) {
  let inThrottle;
  return (...args) => {
    if (!inThrottle) {
      fn(...args);
      inThrottle = setTimeout(() => inThrottle = false, limit);
    }
  };
}
window.addEventListener('scroll', throttle(() => console.log('scrolled'), 200));
📦 Memoization (cache ผลลัพธ์)
const memo = (fn) => {
  const cache = {};
  return (...args) => {
    const key = JSON.stringify(args);
    if (cache[key] === undefined) cache[key] = fn(...args);
    return cache[key];
  };
};
const fib = memo(n => n <= 1 ? n : fib(n-1) + fib(n-2));
🔁 Singleton Pattern (module pattern)
const Singleton = (function() {
  let instance;
  function createInstance() { return new Object("I am the instance"); }
  return { getInstance: () => instance || (instance = createInstance()) };
})();
🎭 Event Emitter (Pub/Sub ง่าย ๆ)
class EventEmitter {
  constructor() { this.events = {}; }
  on(event, listener) { (this.events[event] ||= []).push(listener); }
  emit(event, ...args) { this.events[event]?.forEach(fn => fn(...args)); }
  off(event, listenerToRemove) {
    if (!this.events[event]) return;
    this.events[event] = this.events[event].filter(fn => fn !== listenerToRemove);
  }
}

🚀 Curry function
const curry = fn => {
  const curried = (...args) => args.length >= fn.length ? fn(...args) : (...more) => curried(...args, ...more);
  return curried;
};
const add = (a,b,c) => a+b+c;
const curriedAdd = curry(add);
curriedAdd(1)(2)(3); // 6
